insert into obs(id_poste, time, duration, out_temp, out_temp_omm, windchill, dewpoint, soiltemp,
        humidity, humidity_omm, barometer, barometer_omm, heatindex, pressure, wind, wind_dir, wind10_omm,

       from x2 order by date
    );

-- ------------------
-- materialized view
-- ------------------
create materialized view obs_day as
 select timescaledb_experimental.time_bucket_ng('1 day', time) as time, id_poste, avg(out_temp) as out_temp, avg(out_temp_omm) as out_temp_omm, avg(windchill) as windchill,
 avg(dewpoint) as dewpoint, avg(soiltemp) as soiltemp, avg(humidity) as humidity, avg(humidity_omm) as hunidity_omm, 
 avg(barometer) as barometer, avg(barometer_omm) as barometer_omm, avg(heatindex) as heatindex, avg(pressure) as pressure, avg(wind) as wind, max(wind_gust) as wind_gust,
 avg(wind10) as wind10, avg(wind10_omm) as wind10_omm, sum(rain) as rain, sum(rain_omm) as rain_omm, avg(rain_rate) as rain_rate, max(uv_indice) as max_uv_indice,
 max(radiation) as max_radiation, sum(etp_sum) as etp_sum, avg(in_temp) as in_temp, avg(in_humidity) as in_humidity, avg(rx) as rx, avg(voltage) as voltage
 from obs
 group by 1,2
 order by 1,2
;

1 mesure for 1 station, with n last measures
=============================================
select time, case when step = 0 then 'now' else (-interval)::text END AS metric, moyenne from
(select step, (step||${aggreg:singlequote})::interval as interval from generate_series(0, $steps_histo) g(step)) my_steps
 join lateral
(SELECT
    timescaledb_experimental.time_bucket_ng('15m', o.time + interval)::timestamptz AS time, avg(o.$mesure) AS moyenne
    FROM obs o join poste p on p.id = o.id_poste
    WHERE p.meteor = ${station:singlequote} and
          o.time between $__timeFrom()::timestamptz - interval and $__timeTo()::timestamptz - interval
    GROUP BY 1
    ORDER BY 1
) l on true
order by 1

  avg(barometer_omm) as barometer_omm

  time BETWEEN '2021-04-25T20:39:37.248Z' AND '2022-04-25T20:39:37.248Z'
